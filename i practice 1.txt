//practice 1
import java.util.Scanner;
import java.io.*;
class Node{
    String data;
    int mem;
    Node next;
   
}
class List{
    Node head,tail;
    public void insert(String data,int mem){
        Node n=new Node();
        n.data=data;
        n.mem=mem;
        if(head==null){
            head=n;
            tail=n;
        }else{
            tail.next=n;
            tail=n;
        }
    }
    public void display(){
        Node temp=head;
        if(head==null){
            System.out.println("Software are empty");
        }else{
           while(temp!=null){
               System.out.println(temp.data+" "+temp.mem);
               temp=temp.next;
           }
        }}
        public void min(){
            Node temp=head;
            Node min=head;
            while(temp!=null){
                if(min.mem>temp.mem){
                    min=temp;
                }temp=temp.next;
            }System.out.println(min.data+" "+min.mem);
        }
   
    }

public class Main
{
public static void main(String[] args) {
   List l=new List();
   Scanner sc=new Scanner(System.in);
   int choice;
   do{
       System.out.println("1.Insert new software");
       System.out.println("2.Display new software");
       System.out.println("3.Get the minimum software");
       System.out.println("4.Exit");
       System.out.println("Enter choice");
       choice=sc.nextInt();
       switch(choice){
           case 1:
               System.out.println("Enter the software name:");
               String data=sc.next();
               System.out.println("enter the memory used:");
               int mem=sc.nextInt();
               l.insert(data,mem);
               break;
          case 2:
              System.out.println("Software list:");
              l.display();
              break;
          case 3:
              System.out.println("Minimum memory software is:");
              l.min();
              break;
          case 4:
              System.out.println("Exit");
             
       }
   }while(choice!=4);
}
}
