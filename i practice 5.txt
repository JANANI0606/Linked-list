Program:
//practice 5
import java.util.Scanner;
import java.io.*;
class Node{
    int data;
    Node next;
    Node(int data){
        this.data=data;
    }
}
class List{
    Node head,tail;
    public void insert(int data){
        Node n=new Node(data);
        if(head==null){
            head=n;
            tail=n;
        }else{
            tail.next=n;
            tail=n;
        }
    }
    public void display(){
        Node temp=head;
        if(head==null){
            System.out.println("Stock is empty");
        }else{
           while(temp!=null){
               System.out.println(temp.data+" ");
               temp=temp.next;
           }
        }
    }
    public void delatbegin(){
        head=head.next;
    }}
public class Main
{
public static void main(String[] args) {
   Scanner sc=new Scanner(System.in);
   List l=new List();
   int choice;
   do{
       System.out.println("1.Add goods to be delivered");
       System.out.println("2.List the goods");
       System.out.println("3.Delivered a good");
       System.out.println("3.Exit");
       System.out.println("Enter choice:");
       choice=sc.nextInt();
       switch(choice){
           case 1:
               System.out.println("Enter the stock id:");
               int data=sc.nextInt();
               l.insert(data);
               break;
          case 2:
              System.out.println("Stock list:");
           
              l.display();
              break;
          case 3:
              l.delatbegin();
              System.out.println("Delivered a good");
              System.out.println("Stock list:");
             
              l.display();
              break;
          case 4:
              System.out.println("Exit");
             
       }
   }while(choice!=3);
}
}
